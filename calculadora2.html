function calcular(){ let a = parseFloat(document.getElementById("ancho").value); let b = parseFloat(document.getElementById("alto").value); if(!a || !b){ alert("Ingresá ancho y alto"); return; } // Área en m² let area = (a / 100) * (b / 100); // ========================= // FORMULA PRINCIPAL PRO // ========================= // Carteles chicos: // más caros por m² // // Carteles grandes: // más baratos proporcionalmente // // Sin saltos raros // Nunca baja al crecer // ========================= precioPixel = (area * 700000) + (900000 / Math.sqrt(area + 0.15)); // ========================= // REFERENCIA REAL // ========================= // 200 x 40 aprox // nunca menos de 1.300.000 // ========================= if( area >= 0.75 && area <= 0.95 && precioPixel < 1300000 ){ precioPixel = 1300000; } // ========================= // REDONDEO // ========================= precioPixel = Math.round(precioPixel / 10000) * 10000; // ========================= // MONOCROMATICO // ========================= precioMono = precioPixel / 1.1; precioMono = Math.round(precioMono / 10000) * 10000; // ========================= // TIPO VISUAL // ========================= let tipo = ""; if(area <= 0.7){ tipo = "Cartel chico"; } else if(area <= 2){ tipo = "Cartel mediano"; } else{ tipo = "Cartel grande"; } // ========================= // HTML RESULTADO // ========================= document.getElementById("precioBox").innerHTML = `
${tipo}
Blanco frío / color fijo
$${precioMono.toLocaleString("es-AR")}
Pixel LED app celular
$${precioPixel.toLocaleString("es-AR")}
${a} x ${b} cm · ${area.toFixed(2)} m²
`; generarTexto(); }
